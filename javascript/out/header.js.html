/**
 * save an array to the localstorage
 * @param {string} key 
 * @param {string} value 
 */
function saveArrayToLocalStorage(key, value) {
    localStorage.setItem(key, JSON.stringify(value));
}
/**
 * loade an array from the localstorage
 * @param {string} key 
 */
function loadArrayFromLocalStroage(key) {
    JSON.parse(localStorage.getItem(key))
}
